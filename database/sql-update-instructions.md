Hi there! I have fixed the database table error by adding the `CASCADE` keyword to the drop commands, and updated the inventory schema as you requested with SKU, Barcode, etc.

To answer your first question: Yes! You can already add parts to a service order from the "Workshop Queue" details page. Simply click on a service in the queue, and in the "Parts Used" section, there is an "+ Add Part..." dropdown that lets you select items from the inventory to add to that service (the costs will automatically be added to the final checkout).

Here is the updated SQL script to run in your Supabase SQL Editor:

```sql
DROP TABLE IF EXISTS receipts CASCADE;
DROP TABLE IF EXISTS service_parts CASCADE;
DROP TABLE IF EXISTS workshop_queue CASCADE;
DROP TABLE IF EXISTS invoices CASCADE;
DROP TABLE IF EXISTS customers CASCADE;
DROP TABLE IF EXISTS inventory CASCADE;
DROP TABLE IF EXISTS audit_log CASCADE;
DROP TABLE IF EXISTS stats CASCADE;

-- Create customers table
CREATE TABLE customers (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  name TEXT NOT NULL,
  email TEXT NOT NULL,
  phone TEXT NOT NULL,
  is_member BOOLEAN DEFAULT FALSE,
  member_discount_percentage NUMERIC DEFAULT 0,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc', NOW())
);

-- Create inventory table
CREATE TABLE inventory (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  sku TEXT,
  barcode TEXT,
  product_name TEXT NOT NULL,
  category TEXT,
  subcategory TEXT,
  brand TEXT,
  compatible TEXT,
  cost_price NUMERIC DEFAULT 0,
  selling_price NUMERIC DEFAULT 0,
  stock INTEGER NOT NULL DEFAULT 0,
  location_shelf TEXT,
  supplier TEXT,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc', NOW())
);

-- Create workshop_queue table
CREATE TABLE workshop_queue (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  service_id TEXT NOT NULL,
  customer_name TEXT NOT NULL,
  customer_phone TEXT,
  guitar_brand TEXT NOT NULL,
  guitar_model TEXT NOT NULL,
  serial_number TEXT,
  problem_description TEXT,
  action_taken TEXT,
  status TEXT NOT NULL,
  technician TEXT,
  before_photo_url TEXT,
  after_photo_url TEXT,
  has_down_payment BOOLEAN DEFAULT FALSE,
  down_payment_amount NUMERIC DEFAULT 0,
  estimated_fee NUMERIC NOT NULL,
  final_fee NUMERIC,
  pickup_date TEXT,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc', NOW())
);

-- Create service_parts table (Inventory used in a repair)
CREATE TABLE service_parts (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  service_id BIGINT REFERENCES workshop_queue(id) ON DELETE CASCADE,
  inventory_id BIGINT REFERENCES inventory(id),
  quantity INTEGER NOT NULL,
  price_per_unit NUMERIC NOT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc', NOW())
);

-- Create receipts table
CREATE TABLE receipts (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  service_id BIGINT REFERENCES workshop_queue(id) ON DELETE CASCADE,
  receipt_id TEXT NOT NULL,
  subtotal NUMERIC NOT NULL,
  discount_amount NUMERIC DEFAULT 0,
  tax_amount NUMERIC DEFAULT 0,
  total NUMERIC NOT NULL,
  payment_method TEXT,
  amount_paid NUMERIC NOT NULL,
  change_amount NUMERIC DEFAULT 0,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc', NOW())
);

-- Create audit_log table
CREATE TABLE audit_log (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  time TEXT NOT NULL,
  event TEXT NOT NULL,
  details TEXT NOT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc', NOW())
);

-- Create stats table
CREATE TABLE stats (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  "activeRepairs" INTEGER NOT NULL,
  "activeRepairsDelta" TEXT NOT NULL,
  revenue NUMERIC NOT NULL,
  "lowStock" INTEGER NOT NULL,
  "avgTurnaround" NUMERIC NOT NULL,
  created_at TIMESTAMP WITH TIME ZONE DEFAULT TIMEZONE('utc', NOW())
);
```
